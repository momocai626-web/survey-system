課前 GAS 完整程式碼：
----------------------------------------
const SPREADSHEET_ID = '1Mk65vDEcB2J2lBEqcFZ7wkWrm17MdsoVJZWPd72Dw5E';

function doPost(e) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getActiveSheet();
  const data = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['GitHub','VSCode','Qwen','Pages','Learn','Difficulty','Timestamp']);
  }
  sheet.appendRow([
    data.github || '', data.vscode || '', data.qwen || '', data.pages || '',
    data.learn || '', data.difficulty || '', data.timestamp || new Date().toISOString()
  ]);
  return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getActiveSheet();
  const rows = sheet.getDataRange().getValues();
  const data = rows.slice(1).filter(row => row.some(cell => cell !== ''));
  if (data.length === 0) return ContentService
    .createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);
  const result = data.map(row => ({
    github: row[0]?row[0].toString():'', vscode: row[1]?row[1].toString():'',
    qwen:   row[2]?row[2].toString():'', pages:  row[3]?row[3].toString():'',
    learn:  row[4]?row[4].toString():'', difficulty: row[5]?row[5].toString():'',
    timestamp: row[6]?row[6].toString():''
  }));
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
----------------------------------------

⚠️ 注意：doGet 欄位順序依 Sheets 實際欄序（timestamp 在 row[6]），
   不能用 toLowerCase() 表頭法，因為表頭名稱與 key 不一致。

=====================================
【Google Apps Script 設定重點】
=====================================

部署設定（每個 GAS 都一樣）：
- 類型：網頁應用程式
- 執行身分：我
- 誰可以存取：所有人（含匿名）
- 第一次使用需要授權（點 Review Permissions → Allow）

常見問題：
1. Failed to fetch → 檢查部署設定是否為「所有人」
2. 資料全部 0 → doGet 欄位順序與 Sheets 實際欄序不符，需用 row[index] 對應
3. 只存檔沒更新部署 → 需在「管理部署作業」選新版本才生效
4. 部署成函式庫 → 要選「網頁應用程式」，URL 格式為 .../exec
